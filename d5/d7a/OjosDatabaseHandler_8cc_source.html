171 unsigned long OjosDatabaseHandler::selectOrInsertLicenseIdForName(
172  string rfShortName, const int groupId, const int userId)
173 {
174  bool success = false;
175  unsigned long result = 0;
176 
177  icu::UnicodeString unicodeCleanShortname = fo::recodeToUnicode(rfShortName);
178 
179  // Clean shortname to get utf8 string
180  rfShortName = "";
181  unicodeCleanShortname.toUTF8String(rfShortName);
182 
183  fo_dbManager_PreparedStatement *searchWithOr = fo_dbManager_PrepareStamement(
184  dbManager.getStruct_dbManager(),
185  "selectLicenseIdWithOrOJO",
186  "WITH mainLicense AS ("
187  " SELECT rf_pk, 1 AS prior FROM ONLY license_ref"
188  " WHERE LOWER(rf_shortname) = LOWER($1)"
189  " OR LOWER(rf_shortname) = LOWER($2)"
190  "), candidateLicense AS ("
191  " SELECT rf_pk, 2 AS prior FROM license_candidate"
192  " WHERE (LOWER(rf_shortname) = LOWER($1)"
193  " OR LOWER(rf_shortname) = LOWER($2))"
194  " AND group_fk = $3"
195  ")"
196  "SELECT rf_pk, prior FROM mainLicense "
197  "UNION "
198  "SELECT rf_pk, prior FROM candidateLicense "
199  "ORDER BY prior;",
200  char*, char*, int);
201 
202  /* First check similar matches */
203  /* Check if the name ends with +, -or-later, -only */
204  if (hasEnding(rfShortName, "+") || hasEnding(rfShortName, "-or-later"))
205  {
206  string tempShortName(rfShortName);
207  /* Convert shortname to lower-case */
208  std::transform(tempShortName.begin(), tempShortName.end(), tempShortName.begin(),
209  ::tolower);
210  string plus("+");
211  string orLater("-or-later");
212 
213  unsigned long int plusLast = tempShortName.rfind(plus);
214  unsigned long int orLaterLast = tempShortName.rfind(orLater);
215 
216  /* Remove last occurrence of + and -or-later (if found) */
217  if (plusLast != string::npos)
218  {
219  tempShortName.erase(plusLast, string::npos);
220  }
221  if (orLaterLast != string::npos)
222  {
223  tempShortName.erase(orLaterLast, string::npos);
224  }
225 
226  QueryResult queryResult = dbManager.execPrepared(searchWithOr,
227  (tempShortName + plus).c_str(), (tempShortName + orLater).c_str(),
228  groupId);
229 
230  success = queryResult && queryResult.getRowCount() > 0;
231  if (success)
232  {
233  result = queryResult.getSimpleResults<unsigned long>(0, fo::stringToUnsignedLong)[0];
234  }
235  }
236  else
237  {
238  string tempShortName(rfShortName);
239  /* Convert shortname to lower-case */
240  std::transform(tempShortName.begin(), tempShortName.end(), tempShortName.begin(),
241  ::tolower);
242  string only("-only");
243 
244  unsigned long int onlyLast = tempShortName.rfind(only);
245 
246  /* Remove last occurrence of -only (if found) */
247  if (onlyLast != string::npos)
248  {
249  tempShortName.erase(onlyLast, string::npos);
250  }
251 
252  QueryResult queryResult = dbManager.execPrepared(searchWithOr,
253  tempShortName.c_str(), (tempShortName + only).c_str(), groupId);
254 
255  success = queryResult && queryResult.getRowCount() > 0;
256  if (success)
257  {
258  result = queryResult.getSimpleResults<unsigned long>(0, fo::stringToUnsignedLong)[0];
259  }
260  }
261 
262  if (result > 0)
263  {
264  return result;
265  }
266 
267  if (groupId < 1)
268  {
269  return 0;
270  }
271 
272  if (userId < 1)
273  {
274  return 0;
275  }
276 
277  unsigned count = 0;
278  while ((!success) && count++ < 3)
279  {
280  if (!dbManager.begin())
281  continue;
282 
283  dbManager.queryPrintf("LOCK TABLE license_ref");
284 
285  QueryResult queryResult =
286  dbManager.execPrepared(
287  fo_dbManager_PrepareStamement(dbManager.getStruct_dbManager(),
288  "selectOrInsertLicenseIdForNameOjo",
289  "WITH "
290  "selectExisting AS ("
291  "SELECT rf_pk FROM ONLY license_ref"
292  " WHERE LOWER(rf_shortname) = LOWER($1)"
293  "),"
294  "insertNew AS ("
295  "INSERT INTO license_candidate"
296  "(rf_shortname, rf_text, rf_detector_type, group_fk, "
297  "rf_user_fk_created, rf_user_fk_modified)"
298  " SELECT $1, $2, $3, $4, $5, $5"
299  " WHERE NOT EXISTS(SELECT * FROM selectExisting)"
300  " RETURNING rf_pk"
301  ") "
302 
303  "SELECT rf_pk FROM insertNew "
304  "UNION "
305  "SELECT rf_pk FROM selectExisting",
306  char*, char*, int, int, int
307  ),
308  rfShortName.c_str(), "License by OJO.", 3, groupId, userId);
309 
310  success = queryResult && queryResult.getRowCount() > 0;
311 
312  if (success)
313  {
314  success &= dbManager.commit();
315 
316  if (success)
317  {
318  result = queryResult.getSimpleResults(0, fo::stringToUnsignedLong)[0];
319  }
320  }
321  else
322  {
323  dbManager.rollback();
324  }
325  }
326 
327  return result;
328 }
