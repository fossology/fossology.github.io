66  stringstream tstream;
67  tstream << t.rdbuf();
68  string s = tstream.str();
69  string::size_type pos = 0;
70  string::size_type cpos = 0;

72  {
73  string::size_type tpos = s.find("<s>", pos);
74  if (tpos == string::npos) break;
75  cpos += tpos - pos;
