67  public function handleFileUpload($uploadedFile,$delimiter=',',$enclosure='"')
68  {
69  $errMsg = '';
70  if (! ($uploadedFile instanceof UploadedFile)) {
71  $errMsg = _("No file selected");
72  } elseif ($uploadedFile->getError() !== UPLOAD_ERR_OK) {
73  $errMsg = $uploadedFile->getErrorMessage();
74  } elseif ($uploadedFile->getSize() == 0 && $uploadedFile->getError() == 0) {
75  $errMsg = _("Larger than upload_max_filesize ") .
76  ini_get(self::KEY_UPLOAD_MAX_FILESIZE);
77  } elseif ($uploadedFile->getClientOriginalExtension() != 'csv'
78  && $uploadedFile->getClientOriginalExtension() != 'json') {
79  $errMsg = _('Invalid file extension ') .
80  $uploadedFile->getClientOriginalExtension() . ' of file ' .
81  $uploadedFile->getClientOriginalName();
82  }
83  if (! empty($errMsg)) {
84  return array(false, $errMsg,400);
85  }
87  $licenseCsvImport = $this->getObject('app.license_csv_import');
88  $licenseCsvImport->setDelimiter($delimiter);
89  $licenseCsvImport->setEnclosure($enclosure);
90 
91  return array(true,$licenseCsvImport->handleFile($uploadedFile->getRealPath(), $uploadedFile->getClientOriginalExtension()),200);
92  }
