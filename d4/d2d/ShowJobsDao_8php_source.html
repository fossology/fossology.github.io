53  function uploads2Jobs($upload_pks, $page = 0)
54  {
55  $jobArray = array();
56 
57  // only use the uploads the user / group has access to
58  $upload_pks = array_filter($upload_pks, function($upload_pk) {
59  return $upload_pk !== null && $this->uploadDao->isAccessible($upload_pk, Auth::getGroupId());
60  });
61 
62  if (empty($upload_pks)) {
63  return $jobArray;
64  }
65 
66  $upload_pks = array_values($upload_pks);
67  $uploadList = $this->toPgIntArray($upload_pks);
68 
69  /* Count total jobs for pagination */
70  $countRow = $this->dbManager->getSingleRow(
71  "SELECT count(*) AS cnt FROM job WHERE job_upload_fk = ANY($1)",
72  array($uploadList),
73  __METHOD__ . ".countJobs"
74  );
75  $totalJobCount = intval($countRow['cnt']);
76  $totalPages = floor($totalJobCount / $this->maxJobsPerPage);
77 
78  /* Fetch paginated jobs */
79  $offset = empty($page) ? 0 : $page * $this->maxJobsPerPage;
80  $statementName = __METHOD__ . ".paginatedJobs";
81  $this->dbManager->prepare($statementName,
82  "SELECT job_pk FROM job WHERE job_upload_fk = ANY($1) " .
83  "ORDER BY job_pk DESC LIMIT $2 OFFSET $3");
84  $result = $this->dbManager->execute($statementName,
85  array($uploadList, $this->maxJobsPerPage, $offset));
86 
87  while ($row = $this->dbManager->fetchArray($result)) {
88  $jobArray[] = $row['job_pk'];
89  }
90  $this->dbManager->freeResult($result);
91 
92  return array($jobArray, $totalPages);
93  } /* uploads2Jobs() */

152  }

307  private function attachUploadData(&$jobData, $uploadRows, $uploadtreeRows, $deletedMap)
308  {
309  foreach ($jobData as $job_pk => &$data) {
310  $upload_pk = $data['job']['job_upload_fk'] ?? null;
311  if (empty($upload_pk)) {
312  continue;
313  }
314  if (isset($uploadRows[$upload_pk])) {
315  $data['upload'] = $uploadRows[$upload_pk];
316  $data['uploadtree'] = $uploadtreeRows[$upload_pk] ?? null;
317  } elseif (isset($deletedMap[$job_pk])) {
318  $uploadFk = $deletedMap[$job_pk];
319  $jobName = $this->getJobName($uploadFk);
320  $data['upload'] = array(
321  'upload_filename' => "Deleted Upload: " . $uploadFk . "(" . $jobName . ")",
322  'upload_pk' => $uploadFk,
323  );
324  }
325  }
326  unset($data);
327  }

334  private function attachJobQueueData(&$jobData, $jobPkArray)
335  {
336  $stmt = __METHOD__;
337  $this->dbManager->prepare($stmt,
338  "SELECT jq.*, jd.jdep_jq_depends_fk
339  FROM jobqueue jq
340  LEFT OUTER JOIN jobdepends jd ON jq.jq_pk = jd.jdep_jq_fk
341  WHERE jq.jq_job_fk = ANY($1::int[])
342  ORDER BY jq.jq_pk ASC");
343  $result = $this->dbManager->execute($stmt, array($jobPkArray));
344 
345  foreach ($this->dbManager->fetchAll($result) as $jobQueueRec) {
346  $job_pk = $jobQueueRec['jq_job_fk'];
347  $jq_pk = $jobQueueRec['jq_pk'];
348 
349  if (!isset($jobData[$job_pk])) {
350  continue;
351  }
352 
353  if (isset($jobData[$job_pk]['jobqueue'][$jq_pk])) {
354  $jobData[$job_pk]['jobqueue'][$jq_pk]['depends'][] = $jobQueueRec['jdep_jq_depends_fk'];
355  } else {
356  $jobQueueRec['depends'] = array($jobQueueRec['jdep_jq_depends_fk']);
357  $jobData[$job_pk]['jobqueue'][$jq_pk] = $jobQueueRec;

382  public function getEstimatedTime($job_pk, $jq_Type='', $filesPerSec=0, $uploadId=0, $timeInSec=0)
383  {
384  if (!empty($uploadId)) {
385  $itemCount = $this->dbManager->getSingleRow(
386  "SELECT jq_itemsprocessed FROM jobqueue INNER JOIN job ON jq_job_fk=job_pk "
387  . " WHERE jq_type LIKE 'ununpack' AND jq_end_bits ='1' AND job_upload_fk=$1",
388  array($uploadId),
389  __METHOD__.'.ununpack_might_be_in_other_job'
390  );
391  } else {
392  $itemCount = $this->dbManager->getSingleRow(
393  "SELECT jq_itemsprocessed FROM jobqueue WHERE jq_type LIKE 'ununpack' AND jq_end_bits ='1' AND jq_job_fk =$1",
394  array($job_pk),
395  __METHOD__.'.ununpack_must_be_in_this_job'
396  );
397  }
398 
399  if (!empty($itemCount['jq_itemsprocessed']) && $jq_Type !== 'decider') {
400 
401  $selectCol = "jq_type, jq_endtime, jq_starttime, jq_itemsprocessed";
402  if (empty($jq_Type)) {
403  $removeType = "jq_type NOT LIKE 'ununpack' AND jq_type NOT LIKE 'reportgen' AND jq_type NOT LIKE 'decider' AND jq_type NOT LIKE 'softwareHeritage' AND";
404  /* get starttime endtime and jobtype form jobqueue for a jobid except $removeType */
405  $statementName = __METHOD__."$selectCol.$removeType";
406  $this->dbManager->prepare($statementName,
407  "SELECT $selectCol FROM jobqueue WHERE $removeType jq_job_fk =$1 ORDER BY jq_type DESC");
408  } else {
409  $statementName = __METHOD__."$selectCol.$jq_Type";
410  $this->dbManager->prepare($statementName,
411  "SELECT $selectCol FROM jobqueue WHERE jq_type LIKE '$jq_Type' AND jq_job_fk =$1");
412  }
413  $result = $this->dbManager->execute($statementName, array($job_pk));
414  $estimatedArray = array(); // estimate time for each agent
415 
416  while ($row = $this->dbManager->fetchArray($result)) {
417  $timeOfCompletion = 0;
418  if (empty($row['jq_endtime']) && !empty($row['jq_starttime'])) { // for agent started and not ended
419  if (empty($filesPerSec)) {
420  $burnTime = time() - strtotime($row['jq_starttime']);
421  $filesPerSec = $this->getNumItemsPerSec($row['jq_itemsprocessed'], $burnTime);
422  }
423 
424  if (!empty($filesPerSec)) {
425  $timeOfCompletion = ($itemCount['jq_itemsprocessed'] - $row['jq_itemsprocessed']) / $filesPerSec;
426  }
427  $estimatedArray[] = $timeOfCompletion;
428  }
429  }
430  if (empty($estimatedArray)) {
431  return "";
432  } else {
433  $estimatedTime = round(max($estimatedArray)); // collecting max agent time in seconds
434  if (!empty($timeInSec)) {
435  return intval(!empty($estimatedTime) ? $estimatedTime : 0);
436  }
437  return intval($estimatedTime/3600).gmdate(":i:s", $estimatedTime); // convert seconds to time and return
438  }
439  }
440  }/* getEstimatedTime() */
